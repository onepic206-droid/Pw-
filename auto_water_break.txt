--[[
  Auto Water Break - Multi Bot
  
  Features:
  1. Each bot generates a random world name (7 uppercase letters)
  2. Bot breaks all water blocks in the world
  3. If water is depleted, generate a new world
  4. If inventory water/seed >= threshold, drop to save world
  
  IMPORTANT: Adjust WATER_BLOCK_ID to the correct water block ID!
]]

-- ==================== CONFIG ====================
local CONFIG = {
    SAVE_WORLD      = "yourworldname",  -- World name to drop items (CHANGE THIS)
    DROP_X          = 72,               -- X coordinate for drop in save world
    DROP_Y          = 33,               -- Y coordinate for drop in save world

    WATER_BLOCK_ID  = 18,               -- Block type ID for water (adjust if different)

    DROP_THRESHOLD  = 50,               -- Drop if water block OR seed >= this

    HIT_DELAY       = 300,              -- Delay between hits (ms)
    MAX_HITS        = 25,               -- Max hits per tile
    TILE_DELAY      = 150,              -- Delay between tiles (ms)
    COLLECT_DELAY   = 100,              -- Delay between collects (ms)
    RECONNECT_DELAY = 5000,             -- Delay before reconnect (ms)
    WORLD_LOAD_DELAY = 4000,            -- Delay after warp (ms)
    PATH_TIMEOUT    = 10000,            -- Pathfinding timeout (ms)
    RAND_MIN        = 30,               -- Anti-kick random delay min
    RAND_MAX        = 150,              -- Anti-kick random delay max
}
-- ================================================

-- ==================== UTILS =====================
local function log(botId, msg)
    print("[Bot " .. botId .. "] " .. msg)
end

local function generateWorldName()
    local name = ""
    for i = 1, 7 do
        name = name .. string.char(64 + math.random(1, 26))
    end
    return name
end
-- ================================================

-- ==================== BOT LOGIC =================
function botMain(client, botId)
    local function info(m) log(botId, "[>] " .. m) end
    local function warn(m) log(botId, "[!] " .. m) end

    -- Reconnect helper (auto reconnect on disconnect)
    local function ensureConnected()
        if client:connected() then return true end
        info("Reconnecting...")
        sleep(CONFIG.RECONNECT_DELAY)
        client:connect()
        local t = 0
        while not client:connected() and t < 15000 do
            sleep(1000)
            t = t + 1000
        end
        if not client:connected() then
            warn("Reconnect failed!")
            return false
        end
        info("Reconnected!")
        return true
    end

    -- Wait until world is fully loaded
    local function waitForWorld()
        sleep(CONFIG.WORLD_LOAD_DELAY)
        local t = 0
        while t < 15000 do
            local w = client:world()
            if w then
                local nav = client:navigation()
                if nav and nav ~= "#menu" then return true end
            end
            sleep(500)
            t = t + 500
        end
        return false
    end

    -- Pathfind and walk to a point
    local function pathTo(point)
        if not ensureConnected() then return false end
        local mp = client:point()
        if not mp then return false end
        if mp:equals(point) then return true end
        local ok = client:findPath(point)
        if not ok then return false end
        local t = 0
        while client:pathfinding() and t < CONFIG.PATH_TIMEOUT do
            sleep(100)
            t = t + 100
            if not client:connected() then return false end
        end
        if client:pathfinding() then
            client:clearPath()
            return false
        end
        return true
    end

    -- Check if tile has water
    local function tileHasWater(point)
        local w = client:world()
        if not w then return false end
        local tile = w:tile(point)
        if not tile then return false end
        return (tile.water ~= nil and tile.water ~= 0)
    end

    -- Get all water tiles in the current world
    local function getWaterTiles()
        local w = client:world()
        if not w then return {} end
        local result = {}
        for _, tile in pairs(w.tiles) do
            if tile.water and tile.water ~= 0 then
                table.insert(result, tile:point())
            end
        end
        return result
    end

    -- Break water block at a point
    local function breakWater(point)
        if not tileHasWater(point) then return true end

        -- Walk to an adjacent position near the target
        local mp = client:point()
        if not mp then return false end
        local dx = math.abs(mp.x - point.x)
        local dy = math.abs(mp.y - point.y)
        local inRange = (dx <= 1 and dy <= 1)

        if not inRange then
            local w = client:world()
            if not w then return false end
            local size = w.size
            local spots = {
                Vector2i.new(point.x, point.y + 1),
                Vector2i.new(point.x - 1, point.y),
                Vector2i.new(point.x + 1, point.y),
                Vector2i.new(point.x, point.y - 1),
                Vector2i.new(point.x - 1, point.y + 1),
                Vector2i.new(point.x + 1, point.y + 1),
            }
            local reached = false
            for _, spot in ipairs(spots) do
                if spot.x >= 0 and spot.x < size.x and spot.y >= 0 and spot.y < size.y then
                    if pathTo(spot) then
                        reached = true
                        break
                    end
                end
            end
            if not reached then return false end
        end

        -- Hit the tile until water is removed
        for i = 1, CONFIG.MAX_HITS do
            if not client:connected() then return false end
            client:hit(point)
            sleep(CONFIG.HIT_DELAY + math.random(CONFIG.RAND_MIN, CONFIG.RAND_MAX))
            if not tileHasWater(point) then return true end
        end
        return false
    end

    -- Check inventory water block and seed count
    local function getWaterCounts()
        local inv = client:inventory()
        if not inv then return 0, 0 end
        local blocks = 0
        local seeds = 0
        pcall(function()
            blocks = inv:count(CONFIG.WATER_BLOCK_ID, InventoryItemType.water) or 0
        end)
        pcall(function()
            seeds = inv:count(CONFIG.WATER_BLOCK_ID, InventoryItemType.seed) or 0
        end)
        return blocks, seeds
    end

    -- Check if inventory has reached drop threshold
    local function shouldDrop()
        local blocks, seeds = getWaterCounts()
        return (blocks >= CONFIG.DROP_THRESHOLD or seeds >= CONFIG.DROP_THRESHOLD)
    end

    -- Safely collect dropped items on the ground
    local function safeCollect()
        pcall(function()
            local w = client:world()
            if not w then return end
            local cols = w.collectables
            if not cols then return end
            for _, c in pairs(cols) do
                pcall(function() client:collect(c.id) end)
                sleep(CONFIG.COLLECT_DELAY)
            end
        end)
    end

    -- Drop items at the save world
    local function dropToSaveWorld()
        info("Inventory >= " .. CONFIG.DROP_THRESHOLD .. "! Going to save world...")
        client:warp(CONFIG.SAVE_WORLD)
        if not waitForWorld() then
            warn("Failed to enter save world!")
            return false
        end
        sleep(2000)

        local dropPoint = Vector2i.new(CONFIG.DROP_X, CONFIG.DROP_Y)
        if not pathTo(dropPoint) then
            warn("Failed to path to drop point, trying alternate route...")
            local walked = false
            for ox = -2, 2 do
                for oy = -2, 2 do
                    if ox ~= 0 or oy ~= 0 then
                        local alt = Vector2i.new(CONFIG.DROP_X + ox, CONFIG.DROP_Y + oy)
                        if pathTo(alt) then
                            walked = true
                            break
                        end
                    end
                end
                if walked then break end
            end
            if not walked then
                warn("Cannot reach drop point!")
                return false
            end
        end
        sleep(500)

        local blocks, seeds = getWaterCounts()
        if blocks > 0 then
            info("Dropping " .. blocks .. " water blocks...")
            client:drop(CONFIG.WATER_BLOCK_ID, InventoryItemType.water, blocks)
            sleep(1000)
        end
        if seeds > 0 then
            info("Dropping " .. seeds .. " water seeds...")
            client:drop(CONFIG.WATER_BLOCK_ID, InventoryItemType.seed, seeds)
            sleep(1000)
        end

        info("Drop complete! Resuming farming...")
        sleep(500)
        return true
    end

    -- =========== MAIN BOT LOOP ===========
    if not ensureConnected() then
        warn("Cannot connect!")
        return
    end
    info("Bot started!")

    local totalBroken = 0
    local worldCount = 0

    while true do
        if not ensureConnected() then break end

        -- Check threshold before generating a new world
        if shouldDrop() then
            dropToSaveWorld()
        end

        -- Generate a new random world
        worldCount = worldCount + 1
        local worldName = generateWorldName()
        info("World #" .. worldCount .. ": " .. worldName)
        client:warp(worldName)

        if not waitForWorld() then
            warn("Failed to enter " .. worldName .. ", skipping...")
            sleep(2000)
        else
            info("Entered world: " .. worldName)
            sleep(1000)

            -- Loop: break water blocks in this world
            local worldBroken = 0
            local noProgressCount = 0

            while true do
                if not ensureConnected() then break end

                -- Check drop threshold
                if shouldDrop() then
                    dropToSaveWorld()
                    break -- Exit, generate new world
                end

                -- Find water tiles
                local waterTiles = getWaterTiles()
                if #waterTiles == 0 then
                    info("Water depleted! Broken in this world: " .. worldBroken)
                    break
                end

                info("Found " .. #waterTiles .. " water tiles")
                local brokenThisRound = 0

                for _, point in ipairs(waterTiles) do
                    if not ensureConnected() then break end
                    if shouldDrop() then break end

                    if tileHasWater(point) then
                        local ok = breakWater(point)
                        if ok then
                            worldBroken = worldBroken + 1
                            totalBroken = totalBroken + 1
                            brokenThisRound = brokenThisRound + 1
                            if worldBroken % 5 == 0 then
                                info("  Broken: " .. worldBroken .. " (total: " .. totalBroken .. ")")
                            end
                        end
                    end

                    -- Collect every 8 tiles
                    if worldBroken % 8 == 0 then safeCollect() end
                    sleep(CONFIG.TILE_DELAY)
                end

                -- Check progress
                if brokenThisRound == 0 then
                    noProgressCount = noProgressCount + 1
                    if noProgressCount >= 3 then
                        warn("No progress 3x, skip world...")
                        break
                    end
                else
                    noProgressCount = 0
                end

                safeCollect()
                sleep(500)
            end
        end

        -- Pause before next world
        info("Total broken: " .. totalBroken .. " | Worlds: " .. worldCount)
        sleep(1000)
    end
end
-- ================================================

-- ==================== STARTUP ===================
print("============================================")
print("  AUTO WATER BREAK - Multi Bot")
print("  Save World : " .. CONFIG.SAVE_WORLD)
print("  Drop Point : (" .. CONFIG.DROP_X .. ", " .. CONFIG.DROP_Y .. ")")
print("  Threshold  : " .. CONFIG.DROP_THRESHOLD)
print("============================================")

local allClients = {}
for _, c in pairs(getClients()) do
    table.insert(allClients, c)
end

-- If no clients found, use the current client
if #allClients == 0 then
    table.insert(allClients, getClient())
end

print("[*] Found " .. #allClients .. " bot(s)")

for i, c in ipairs(allClients) do
    local botIndex = i
    runThread(function()
        sleep(botIndex * 2000) -- Stagger start between bots
        log(botIndex, "Starting...")
        local ok, e = pcall(botMain, c, botIndex)
        if not ok then
            log(botIndex, "[X] CRASH: " .. tostring(e))
        end
        log(botIndex, "Bot finished.")
    end)
end
print("[*] All bot threads started!")
-- ================================================
